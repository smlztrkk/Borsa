const DovizSil = async (dovizToDelete) => {
    setIsLoading(true);

    // Belge referansını tanımlayın
    const docRef = doc(db, "Portfolio", "EsuOqP155Qo3YdAdwK9y");

    try {
        // Mevcut belge verilerini alın
        const docSnapshot = await getDoc(docRef);
        let existingData = {};

        if (docSnapshot.exists()) {
            existingData = docSnapshot.data();
        }

        // Mevcut döviz verilerini alın
        let existingDoviz = existingData.Döviz || [];

        // Mevcut döviz verileri içinden silinecek döviz verilerini çıkarın
        const updatedDoviz = existingDoviz.filter(doviz => {
            return !dovizToDelete.includes(doviz);
        });

        // Güncellenmiş döviz verilerini belgeye güncelleyin
        await updateDoc(docRef, { Döviz: updatedDoviz });

        console.log("Başarılı döviz silme.");
        setIsLoading(false);
    } catch (error) {
        console.error("Belgeyi güncellerken hata oluştu:", error);
        setIsLoading(false);
    }
};
